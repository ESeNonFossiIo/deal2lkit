171 template <int dim>
172 void SerialLaplace<dim>::setup_system ()
173 {
174  dof_handler->distribute_dofs (*fe);
175  std::cout << "Number of degrees of freedom: "
176  << dof_handler->n_dofs()
177  << std::endl;
178 
179  constraints.clear ();
180 
181  DoFTools::make_hanging_node_constraints (*dof_handler, constraints);
182 
183  dirichlet_bcs.interpolate_boundary_values(*dof_handler,constraints);
184 
185  constraints.close ();
186 
187  DynamicSparsityPattern d_sparsity(dof_handler->n_dofs());
188 
189  DoFTools::make_sparsity_pattern (*dof_handler, d_sparsity,
190  constraints);
191 
192  sparsity_pattern.copy_from(d_sparsity);
193 
194  system_matrix.reinit (sparsity_pattern);
195 
196  solution.reinit (dof_handler->n_dofs());
197 
198  system_rhs.reinit (dof_handler->n_dofs());
199 }

239 template <int dim>
240 void SerialLaplace<dim>::refine_grid ()
241 {
242  if (refinement_mode == "global_refinement")
243  {
244  triangulation->refine_global (1);
245  }
246  else if (refinement_mode == "adaptive_refinement")
247  {
248  Vector<float> estimated_error_per_cell (triangulation->n_active_cells());
249 
250  KellyErrorEstimator<dim>::estimate (*dof_handler,
251  QGauss<dim-1>(3),
252  typename FunctionMap<dim>::type(),
253  solution,
254  estimated_error_per_cell);
255 
256  GridRefinement::refine_and_coarsen_fixed_number (*triangulation,
257  estimated_error_per_cell,
258  0.3, 0.03);
259 
260  triangulation->execute_coarsening_and_refinement ();
261  }
262  else
263  {
264  Assert (false, ExcNotImplemented());
265  }
266 }

268 template <int dim>
269 void SerialLaplace<dim>::run ()
270 {
271  make_grid_fe ();
272  for (unsigned int cycle=0; cycle<n_cycles; ++cycle)
273  {
274  std::cout<<"cycle : "<<cycle+1<<std::endl;
275  setup_system ();
276  assemble_system ();
277  solve ();
278  process_solution (cycle);
279  if (cycle < n_cycles-1)
280  refine_grid ();
281  else
282  output_results ();
283  }
284  eh.output_table(std::cout);
285 
286 }
