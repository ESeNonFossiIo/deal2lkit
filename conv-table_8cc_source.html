111  template <int dim>
112  double Solution<dim>::value (const Point<dim> &p,
113  const unsigned int) const
114  {
115  double return_value = 0;
116  for (unsigned int i=0; i<this->n_source_centers; ++i)
117  {
118  const Tensor<1,dim> x_minus_xi = p - this->source_centers[i];
119  return_value += std::exp(-x_minus_xi.norm_square() /
120  (this->width * this->width));
121  }
122 
123  return return_value;
124  }

128  Tensor<1,dim> Solution<dim>::gradient (const Point<dim> &p,
129  const unsigned int) const

131  Tensor<1,dim> return_value;
132 
133  for (unsigned int i=0; i<this->n_source_centers; ++i)
134  {
135  const Tensor<1,dim> x_minus_xi = p - this->source_centers[i];
136 
137  return_value += (-2 / (this->width * this->width) *
138  std::exp(-x_minus_xi.norm_square() /
139  (this->width * this->width)) *
140  x_minus_xi);
141  }
142

161  double RightHandSide<dim>::value (const Point<dim> &p,
162  const unsigned int) const
163  {
164  double return_value = 0;
165  for (unsigned int i=0; i<this->n_source_centers; ++i)
166  {
167  const Tensor<1,dim> x_minus_xi = p - this->source_centers[i];
168 
169  return_value += ((2*dim - 4*x_minus_xi.norm_square()/
170  (this->width * this->width)) /
171  (this->width * this->width) *
172  std::exp(-x_minus_xi.norm_square() /
173  (this->width * this->width)));
174  return_value += std::exp(-x_minus_xi.norm_square() /
175  (this->width * this->width));
176  }
177 
178  return return_value;

282  template <int dim>
283  void HelmholtzProblem<dim>::assemble_system ()
284  {
285  QGauss<dim> quadrature_formula(3);
286  QGauss<dim-1> face_quadrature_formula(3);
287 
288  const unsigned int n_q_points = quadrature_formula.size();
289  const unsigned int n_face_q_points = face_quadrature_formula.size();
290 
291  const unsigned int dofs_per_cell = fe->dofs_per_cell;
292 
293  FullMatrix<double> cell_matrix (dofs_per_cell, dofs_per_cell);
294  Vector<double> cell_rhs (dofs_per_cell);
295 
296  std::vector<types::global_dof_index> local_dof_indices (dofs_per_cell);
297 
298  FEValues<dim> fe_values (*fe, quadrature_formula,
299  update_values | update_gradients |
300  update_quadrature_points | update_JxW_values);
301 
302  FEFaceValues<dim> fe_face_values (*fe, face_quadrature_formula,
303  update_values | update_quadrature_points |
304  update_normal_vectors | update_JxW_values);
305 
306  const RightHandSide<dim> right_hand_side;
307  std::vector<double> rhs_values (n_q_points);
308 
309  const Solution<dim> exact_solution;
310 
311  typename DoFHandler<dim>::active_cell_iterator
312  cell = dof_handler->begin_active(),
313  endc = dof_handler->end();
314  for (; cell!=endc; ++cell)
315  {
316  cell_matrix = 0;
317  cell_rhs = 0;
318 
319  fe_values.reinit (cell);
320 
321  right_hand_side.value_list (fe_values.get_quadrature_points(),
322  rhs_values);
323 
324  for (unsigned int q_point=0; q_point<n_q_points; ++q_point)
325  for (unsigned int i=0; i<dofs_per_cell; ++i)
326  {
327  for (unsigned int j=0; j<dofs_per_cell; ++j)
328  cell_matrix(i,j) += ((fe_values.shape_grad(i,q_point) *
329  fe_values.shape_grad(j,q_point)
330  +
331  fe_values.shape_value(i,q_point) *
332  fe_values.shape_value(j,q_point)) *
333  fe_values.JxW(q_point));
334 
335  cell_rhs(i) += (fe_values.shape_value(i,q_point) *
336  rhs_values [q_point] *
337  fe_values.JxW(q_point));
338  }
339 
340  for (unsigned int face_number=0; face_number<GeometryInfo<dim>::faces_per_cell; ++face_number)
341  if (cell->face(face_number)->at_boundary()
342  &&
343  (cell->face(face_number)->boundary_id() == 1))
344  {
345  fe_face_values.reinit (cell, face_number);
346 
347  for (unsigned int q_point=0; q_point<n_face_q_points; ++q_point)
348  {
349  const double neumann_value
350  = (exact_solution.gradient (fe_face_values.quadrature_point(q_point)) *
351  fe_face_values.normal_vector(q_point));
352 
353  for (unsigned int i=0; i<dofs_per_cell; ++i)
354  cell_rhs(i) += (neumann_value *
355  fe_face_values.shape_value(i,q_point) *
356  fe_face_values.JxW(q_point));
357  }
358  }
359 
360  cell->get_dof_indices (local_dof_indices);
361  for (unsigned int i=0; i<dofs_per_cell; ++i)
362  {
363  for (unsigned int j=0; j<dofs_per_cell; ++j)
364  system_matrix.add (local_dof_indices[i],
365  local_dof_indices[j],
366  cell_matrix(i,j));
367 
368  system_rhs(local_dof_indices[i]) += cell_rhs(i);
369  }
370  }
371 
372  hanging_node_constraints.condense (system_matrix);
373  hanging_node_constraints.condense (system_rhs);
374 
375  std::map<types::global_dof_index,double> boundary_values;
376  VectorTools::interpolate_boundary_values (*dof_handler,
377  0,
378  Solution<dim>(),
379  boundary_values);
380  MatrixTools::apply_boundary_values (boundary_values,
381  system_matrix,
382  solution,
383  system_rhs);
384  }

452  template <int dim>
453  void HelmholtzProblem<dim>::run ()
454  {
455  const unsigned int n_cycles = (refinement_mode==global_refinement)?5:9;
456  for (unsigned int cycle=0; cycle<n_cycles; ++cycle)
457  {
458  if (cycle == 0)
459  {
460  make_grid_fe();
461  triangulation->refine_global (3);
462 
463  typename Triangulation<dim>::cell_iterator
464  cell = triangulation->begin (),
465  endc = triangulation->end();
466  for (; cell!=endc; ++cell)
467  for (unsigned int face_number=0;
468  face_number<GeometryInfo<dim>::faces_per_cell;
469  ++face_number)
470  if ((std::fabs(cell->face(face_number)->center()(0) - (-1)) < 1e-12)
471  ||
472  (std::fabs(cell->face(face_number)->center()(1) - (-1)) < 1e-12))
473  cell->face(face_number)->set_boundary_id (1);
474  }
475  else
476  refine_grid ();
477 
478 
479  setup_system ();
480 
481  assemble_system ();
482  solve ();
483 
484  process_solution (cycle);
485  }
486 
487  eh.output_table(std::cout);
488 
489  std::string vtk_filename;
490  switch (refinement_mode)
491  {
492  case global_refinement:
493  vtk_filename = "solution-global";
494  break;
495  case adaptive_refinement:
496  vtk_filename = "solution-adaptive";
497  break;
498  default:
499  Assert (false, ExcNotImplemented());
500  }
501 
502  switch (fe->degree)
503  {
504  case 1:
505  vtk_filename += "-q1";
506  break;
507  case 2:
508  vtk_filename += "-q2";
509  break;
510 
511  default:
512  Assert (false, ExcNotImplemented());
513  }
514 
515  vtk_filename += ".vtk";
516  std::ofstream output (vtk_filename.c_str());
517 
518  DataOut<dim> data_out;
519  data_out.attach_dof_handler (*dof_handler);
520  data_out.add_data_vector (solution, "solution");
521 
522  data_out.build_patches (fe->degree);
523  data_out.write_vtk (output);
524 
525 
526  }
